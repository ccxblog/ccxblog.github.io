java网络编程

发表于 2021-03-29 更新于 2021-03-30 分类于 java

网络编程的目的

直接或间接地通过网络协议与其它计算机实现数据交换进行通讯。

网络编程中的两个问题

  1. 如何准确地定位网络上的一台或多台主机；以及定位主机上特定的应用。

    • 通过 IP 和 端口号

  2. 找到主机后如何可靠高效地进行数据传输。

    • 提供网络通信协议

IP

用来唯一的标识互联网上的一台主机 ， 在 java 中使用 InetAddress 类代表 IP

Java 中 实例化 InetAddress

  1. getByName(String name) 方法 : name 可以是 IP 也 可以是 域名
  2. getLocalHost() 方法 : 获取本机 IP

InetAddress 的两个常用方法

  1. getHostName() : 获取域名
  2. getHostAddress() : 获取 IP 地址
import java.net.InetAddress;
import java.net.UnknownHostException;

public class InetAddressTest {
    public static void main(String[] args) {
        try {
            // 百度 的 ip
            InetAddress inet = InetAddress.getByName("14.215.177.39");
            InetAddress inet2 = InetAddress.getByName("www.baidu.com");
            // 获取本地 ip
            InetAddress inet3 = InetAddress.getLocalHost();

            System.out.println(inet);
            System.out.println(inet2);
            System.out.println(inet3);

            System.out.println(inet.getHostName());
            System.out.println(inet2.getHostAddress());

        } catch (UnknownHostException e) {
            e.printStackTrace();
        }
    }
}

端口号

端口号用来标识计算机上正在运行的进程 ， 要求不同的进程要有不同的端口号 ， 范围 是 0 - 65535

IP 地址 和 端口号 组成一个网络套接字 ： Socket

服务端

@Test
   public void Server(){
       ServerSocket ss = null;
       Socket socket = null;
       InputStreamReader isr = null;
       try {
           // 创建一个端口为 8899 的服务
           ss = new ServerSocket(8899);
           // 获取 客户端请求的 socket
           socket = ss.accept();
           // 获取 输入流
           InputStream is = socket.getInputStream();
           isr = new InputStreamReader(is);
           
           // 接收客户端发送的数据
           int len;
           char[] buff = new char[1024];
           while((len = isr.read(buff)) != -1){
               System.out.println(new String(buff , 0 , len));
           }

       } catch (IOException e) {
           e.printStackTrace();
       }
       finally {
           if(isr != null) {
               try {
                   isr.close();
               } catch (IOException e) {
                   e.printStackTrace();
               }
           }
           if(socket != null) {
               try {
                   socket.close();
               } catch (IOException e) {
                   e.printStackTrace();
               }
           }
           if(ss != null) {
               try {
                   ss.close();
               } catch (IOException e) {
                   e.printStackTrace();
               }
           }
       }
   }

  • 发送文件后，服务器返回一条接收完毕的信息给客户端

客户端

@Test
    public void client() throws IOException {

        InetAddress inet = InetAddress.getLocalHost();
        Socket socket = new Socket(inet , 8899);
        // 获取传输流
        OutputStream os = socket.getOutputStream();
        // 打开本地文件
        FileInputStream fis = new FileInputStream("hello.txt");

        byte c[] = new byte[1024];
        int len;
        // 从本地读入，然后写到服务器上
        while((len = fis.read(c)) != -1){
            os.write(c , 0 , len);
        }

        // 表示停止写入 ， 叫服务器读完就不要读了
        socket.shutdownOutput();

        InputStream is = socket.getInputStream();

        // 字节缓存流 ， 先把一次读入的数据写到缓存里面，最后一起输出
        ByteArrayOutputStream bis = new ByteArrayOutputStream();
        byte cc[] = new byte[1024];
        int len1 = 0 , s = 0;
        while ((len1 = is.read(cc)) != -1){
            bis.write(cc , 0 , len1);
            s += len1;
        }

        System.out.println(bis.toString());

        bis.close();
        is.close();
        fis.close();
        os.close();
        socket.close();
    }

UDP

  • UPD 实现数据的发送

发送方

@Test
   public void sender() throws IOException {
       DatagramSocket socket = new DatagramSocket();
       String s = "向你发送一条数据";
       InetAddress inet = InetAddress.getLocalHost();
       // 数据 ， 数据长度 ， 地址 ， 端口
       DatagramPacket data =
               new DatagramPacket(s.getBytes() , 0 , s.getBytes().length , inet , 8899);
       // 发送
       socket.send(data);
       // 关闭
       socket.close();
   }

接收方

@Test
    public void reve() throws IOException {

        // 设置端口
        DatagramSocket socket = new DatagramSocket(8899);
        // 设置一次接收的容量？？
        byte c[] = new byte[1024];
        DatagramPacket packet = new DatagramPacket(c , 0 , c.length);
        // 接收
        socket.receive(packet);
        // 获取数据和获取长度
        String s = new String(packet.getData() , 0 ,packet.getLength());

        System.out.println(s);
        socket.close();

    }

@Test
public void url() throws IOException {
    // 传入一个要下载 url
    URL url = new URL("https://p1.ssl.qhimg.com/t01740f70820ed293ca.jpg");
    // 获取连接
    HttpURLConnection urlConnection = (HttpURLConnection)url.openConnection();

    // 获取输入流
    InputStream is = urlConnection.getInputStream();
    FileOutputStream fos = new FileOutputStream("baidu.jpg");
    int len;
    byte[] c = new byte[1024];
    while((len = is.read(c)) != -1){
        fos.write(c , 0 , len);
    }

    fos.close();
    is.close();
    // 断开连接
    urlConnection.disconnect();
}
