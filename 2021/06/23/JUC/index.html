JUC

JUC 是 java.util.concurrent 的简称 其主要由三大包构成

  1. java.util.concurrent : java 并发包
  2. java.util.concurrent.atomic : java 并发原子包
  3. java.util.concurrent.locks : java并发锁包
  • synchronized
  1. synchronized 是java中的关键字，发生异常的时候会自动释放锁，所以不会造成死锁。
  2. synchronized 是非公平锁（公平锁一般会维护一个队列，线程有序排队，而非公平锁则会出现插队，插队失败再排队）。
  3. synchronized 修饰静态方法 或者 同步代码块中为 xx.class的时候会锁住 xx.class类的所有实例。
  4. synchronized 修饰普通方法只锁当前实例。
  5. synchronized 修饰同步代码块的时候锁的是代码块里的对象。
  6. synchronized 加锁和解锁都是自动执行。

。。。。

所以可以得出 synchronized 是重量级别的锁。

synchronized 实现生产者消费者

package com.company;

class Ticket{
    private int ticket = 0;
    public synchronized void incr() throws InterruptedException {
        while (ticket != 0) this.wait();
        ticket ++;
        System.out.println(Thread.currentThread().getName() + ticket);
        this.notifyAll();
    }

    public synchronized void decr() throws InterruptedException {
        while (ticket == 0) this.wait();
        ticket --;
        System.out.println(Thread.currentThread().getName() + ticket);
        this.notifyAll();
    }
}


public class Main {

    public static void main(String[] args) {
        Ticket ticket = new Ticket();

        new Thread(()->{
            try {
                for (int i = 0 ; i < 10 ; i ++) {
                    ticket.incr();
                }
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        } , "A").start();


        new Thread(()->{
            try {
                for (int i = 0 ; i < 10 ; i ++)
                ticket.decr();
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        } , "B").start();

        new Thread(()->{
            try {
                for (int i = 0 ; i < 10 ; i ++)
                ticket.incr();
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        } , "C").start();

        new Thread(()->{
            try {
                for (int i = 0 ; i < 10 ; i ++)
                    ticket.decr();
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        } , "D").start();

    }
}

Lock 接口

  • ReentrantLock ( 可重入锁 )
  1. ReentrantLock 可以是公平的也可以是非公平的，默认是非公平的，它有一个构造方法传入 true 的时候是公平锁。
  2. ReentrantLock 锁的是自己本身。
  3. ReentrantLock 需要手动释放锁。

ReentrantLock 实现卖票程序

package com.company;


import java.util.concurrent.locks.Lock;
import java.util.concurrent.locks.ReentrantLock;

class Ticket{
    private int ticket = 30;
    private Lock lock = new ReentrantLock();
    public void sale(){
        lock.lock();
        try {
            if (ticket > 0)
                System.out.println(Thread.currentThread().getName() + "卖出了第" + ticket -- + "票");
        }catch (Exception e){
            e.printStackTrace();
        }finally {
            lock.unlock();
        }

    }
}


public class Main {

    public static void main(String[] args) {
        Ticket ticket = new Ticket();
        new Thread(()->{for (int i = 0 ; i < 40 ; i ++)ticket.sale();} , "A").start();
        new Thread(()->{for (int i = 0 ; i < 40 ; i ++)ticket.sale();} , "B").start();
        new Thread(()->{for (int i = 0 ; i < 40 ; i ++)ticket.sale();} , "C").start();
    }
}

CountDownLatch 守护线程

package com.company;

import java.util.concurrent.CountDownLatch;

class Ticket{
    public void incr() {
        System.out.println(Thread.currentThread().getName() + "执行完了");
    }
}

public class Main {

    public static void main(String[] args) throws InterruptedException {
        Ticket ticket = new Ticket();
        // 设置为 6 个 线程守护
        CountDownLatch countDownLatch = new CountDownLatch(6);
        for (int i = 0 ; i < 6 ; i++){
            new Thread(()->{
                    ticket.incr();
                    countDownLatch.countDown(); // 执行完少一个
            } , String.valueOf(i)).start();
        }
        countDownLatch.await(); // 等待6个线程执行完
        System.out.println("主线程执行完了-----");
    }
}

  • ReadWriteLock (读写锁)
package com.company;

import java.util.HashMap;
import java.util.Map;
import java.util.concurrent.locks.ReadWriteLock;
import java.util.concurrent.locks.ReentrantReadWriteLock;

class MyCache{
    private volatile Map<String , Object> map = new HashMap<>();

    private ReadWriteLock readWriteLock = new ReentrantReadWriteLock();

    public void put(String key , Object value){
        readWriteLock.writeLock().lock();

        try{
            System.out.println(Thread.currentThread().getName() + "正在插入数据");
            map.put(key , value);
            System.out.println(Thread.currentThread().getName() + "数据插入完成");
        }finally {
            readWriteLock.writeLock().unlock();
        }

    }

    public void get(String key){
        readWriteLock.readLock().lock();
        try {
            System.out.println(Thread.currentThread().getName() + "正在读取数据");
            Object o = map.get(key);
            System.out.println(Thread.currentThread().getName() + "读取数据完成");
        }finally {
            readWriteLock.readLock().unlock();
        }
    }

}

public class Test {
    public static void main(String[] args) {
        MyCache cache= new MyCache();

        for (int i = 0 ; i < 5 ; i ++){
            final int k = i;
            new Thread(()->{
                cache.put(String.valueOf(k) , k);
            } , String.valueOf(i)).start();
        }

        for (int i = 0 ; i < 5 ; i ++){
            final int k = i; // 保存在常量池，避免主线程结束后释放资源，其它地方还有用。
            new Thread(()->{
                cache.get(k+"");
            } , String.valueOf(i)).start();
        }
    }
}
  • BlockingQueue（阻塞队列）

队列空的时候 ， 取操作会阻塞， 队列满的时候进队列操作会阻塞，有了BlockingQueue我们不用关系什么时候阻塞线程，什么时候唤醒线程了。

package com.company;

import java.util.concurrent.ArrayBlockingQueue;
import java.util.concurrent.BlockingQueue;
import java.util.concurrent.TimeUnit;

public class MyBlockingQueue {
    public static void main(String[] args) throws InterruptedException {

        // 设置阻塞队列长度
        BlockingQueue<String> blockingQueue = new ArrayBlockingQueue<>(1);

        blockingQueue.add("a");
        // 队列满后 add 方法会报异常
        // blockingQueue.add("d");

        blockingQueue.remove();
        // 队列为空 调用 remove 方法会抛出异常
        // blockingQueue.remove();

        blockingQueue.offer("a");
        // 队列满后调用 remove 方法会返回 false
        // blockingQueue.offer("a");

        blockingQueue.poll();
        // 队列为空调用 poll 方法会放回 false
        // blockingQueue.poll();

        blockingQueue.put("a");
        // 队列满后调用 put 方法会一直阻塞
        // blockingQueue.put("a");

        blockingQueue.take();
        // 队列为空调用 take 方法会一直阻塞
        // blockingQueue.take();

        blockingQueue.offer("a" , 3L , TimeUnit.SECONDS);
        // 队列满后会阻塞 3 秒
        blockingQueue.offer("a" , 3L , TimeUnit.SECONDS);

        blockingQueue.poll(3L , TimeUnit.SECONDS);
        // 队列为空 会阻塞 3 秒
        blockingQueue.poll(3L , TimeUnit.SECONDS);

    }
}

public ThreadPoolExecutor(int corePoolSize, // 核心线程数
                             int maximumPoolSize,  // 最大线程数
                             long keepAliveTime,  // 最大线程存活时间
                             TimeUnit unit,  // 最大线程数存活时间单位
                             BlockingQueue<Runnable> workQueue, // 阻塞队列
                             ThreadFactory threadFactory, // 创建线程的工厂
                             RejectedExecutionHandler handler) { // 拒绝策略
       if (corePoolSize < 0 ||
           maximumPoolSize <= 0 ||
           maximumPoolSize < corePoolSize ||
           keepAliveTime < 0)
           throw new IllegalArgumentException();
       if (workQueue == null || threadFactory == null || handler == null)
           throw new NullPointerException();
       this.acc = System.getSecurityManager() == null ?
               null :
               AccessController.getContext();
       this.corePoolSize = corePoolSize;
       this.maximumPoolSize = maximumPoolSize;
       this.workQueue = workQueue;
       this.keepAliveTime = unit.toNanos(keepAliveTime);
       this.threadFactory = threadFactory;
       this.handler = handler;
   }

  • 四种拒绝策略
    1. new ThreadPoolExecutor.AbortPolicy() ：默认的拒绝策略 ， 到达最大线程数 并且阻塞队列满后抛出异常。
    2. new ThreadPoolExecutor.CallerRunsPolicy() : 让调用线程池的线程去做任务。
    3. new ThreadPoolExecutor.DiscardOldestPolicy() : 丢弃队列最前面的任务，然后重新提交被拒绝的任务
    4. new ThreadPoolExecutor.DiscardPolicy():丢弃任务，但是不抛出异常。
